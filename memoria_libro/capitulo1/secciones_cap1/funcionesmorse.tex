
\section{El Lema de Morse}

Comenzamos introduciendo los puntos crﾃｭticos de una funciﾃｳn suave y el concepto de ﾃｭndice de un punto crﾃｭtico no degenerado. Concluiremos la secciﾃｳn con el primer pilar de la teorﾃｭa de Morse y sobre lo que se edificarﾃ｡n gran parte de los resultados posteriores: el lema de Morse.

\begin{definition}
    Sea $f: \mathrm{M} \to \mathbb{R}$ una funciﾃｳn suave.
    \begin{enumerate}[label=\roman*)]
        \item Diremos que $p\in \mathrm{M}$ es un \textit{punto crﾃｭtico de $f$} si $d_{p}f:T_{p}M\to \mathbb{R}$ es idﾃｩnticamente cero. Es decir, dado un sistema de coordenadas local $(x_{1}, \dots ,x_{n})$ en un entorno $\mathcal{U}$ de $p$ tenemos
        \begin{equation*}
            \frac{\partial f}{\partial x_1}(p)= \dots = \frac{\partial f}{\partial x_n}(p) = 0.
        \end{equation*}
        
        \item A su vez, diremos que $f(p)\in \mathbb{R}$ es un \textit{valor crﾃｭtico de $f$} si $p \in \mathrm{M}$ es un punto crﾃｭtico.
        
    \end{enumerate}
\end{definition}
\bigskip

\begin{definition}
    Sea $p\in \mathrm{M}$ un punto crﾃｭtico de f , diremos que $p$ es \textit{no degenerado} si la matriz  $\left(\frac{\partial^{2}f}{\partial x_i \partial x_j}(p)\right)$ es no singular, es decir, su determinante es no nulo.
\end{definition}
\bigskip

\begin{definition}
    Sean $f:\mathrm{M}\to \mathbb{R}$ una funciﾃｳn suave y $p\in M$ un punto crﾃｭtico de f. Definimos $\textrm{Hess}_{p}f: T_{p}\mathrm{M}\to \mathbb{R}$ como la forma bilineal simﾃｩtrica sobre el plano tangente a $\mathrm{M}$ en $p$ construida como sigue:
    \begin{quote}
        Tomamos $v,w\in T_{p}\mathrm{M}$ y sean $\tilde{v}$ y $\tilde{w}$ extensiones locales suyas a campos vectoriales. Entonces 
        \begin{equation*}
            \text{Hess}_{p}f(v,w)=\tilde{v}_{p}(\tilde{w}_{p}(f)),
        \end{equation*}
        donde $\tilde{v}_{p}$ es simplemente $v$ y $\tilde{w}_{p}(f)$ denota la derivada parcial de $f$ en la direcciﾃｳn de $\tilde{w}$.
    \end{quote}
\end{definition}
\bigskip
\begin{remark}
    Es rﾃ｡pido comprobar que $\text{Hess}_{p}f$ es en efecto simﾃｩtrica y estﾃ｡ bien definida:

    \begin{enumerate}[label=(\roman*)]
        \item Es simﾃｩtrica ya que $\tilde{v}_{p}(\tilde{w}_{p}(f)) - \tilde{w}_{p}(\tilde{v}_{p}(f)) = [\tilde{v},\tilde{w}]_{p}(f) = 0$ donde $[\tilde{v},\tilde{w}]$ es el corchete de Lie de $\tilde{v}$ y $\tilde{w}$ cuya expresiﾃｳn en coordenadas recordemos que es
        \begin{equation*}
            [\tilde{v},\tilde{w}]_{p}(f)=\sum_{i=1}^n \big( \tilde{v}_p(\tilde{w}_{i}) - \tilde{w}_p(\tilde{v}_{i}) \big) \left. \frac{\partial}{\partial x_{i}}\right|_p f.
        \end{equation*}
        Luego se anula al ser $p$ un punto crﾃｭtico.

        \item Por otro lado, estﾃ｡ bien definida ya que $\tilde{v}_{p}(\tilde{w}_{p}(f))$ no depende de las extensiones de $v$ y $w$ elegidas puesto que nos importa el vector del campo en el punto $p$ y este no cambia.
    \end{enumerate}
\end{remark}
\bigskip

Veamos ahora como es $\text{Hess}_pf$ en coordenadas. De nuevo, tomemos un sistema de coordenadas local $(x_{1}, \dots ,x_{n})$, tenemos
\begin{equation*}
    v=\sum_{i=0}^n a_i \left.\frac{\partial}{\partial x_i}\right|_p ~~\text{y}~~ w=\sum_{i=0}^n b_i \left.\frac{\partial}{\partial x_i}\right|_p .
\end{equation*}

Y para las extensiones $\tilde{v}$ y $\tilde{w}$ basta tomar como coeficientes las funciones constantes $a_i$ y $b_j$. Asﾃｭ
\begin{equation*}
    \text{Hess}_pf= \tilde{v}_{p}(\tilde{w}_{p}(f)) = v\left(\sum_{i=0}^n b_i \left.\frac{\partial}{\partial x_i}\right|_p f\right)=\sum_{i,j} a_ib_i\left.\frac{\partial^{2}}{\partial x_i \partial x_j}\right|_p f .
\end{equation*}

Luego la matriz $\left(\frac{\partial^{2}f}{\partial x_i \partial x_j}(p)\right)$ mencionada antes representa la forma bilineal $\text{Hess}_pf$ respecto de la base $\left( \left.\frac{\partial}{\partial x_i}\right|_p,\dots, \left.\frac{\partial}{\partial x_n}\right|_p \right)$ de $T_p\mathrm{M}$.
\bigskip

\begin{definition}
    Sean $V$ un espacio vectorial y $H$ una forma bilineal sobre $V$. Entonces
    
    \begin{enumerate}[label=\roman*)]
        \item El \textit{ﾃｭndice} de $H$ es la mﾃ｡xima dimensiﾃｳn de un subespacio de $V$ sobre el cual $H$ es definida negativa.
        
        \item La \textit{nulidad} de $H$ es la dimensiﾃｳn del radical de $H$, es decir,
        \begin{equation*}
            \text{dim}(\text{Rad}(H)) ~\text{ con }~ \text{Rad}(H)=\{v\in V:H(v,w)=0, ~\forall w\in V \}.
        \end{equation*}
        
    \end{enumerate}
\end{definition}

\begin{remark}
    Un punto crﾃｭtico $p\in M$ de $f$ suave es no degenerado si y solo si la nulidad de $H$ sobre $T_pM$ es $0$. En efecto:
    \vspace{-2em}
    \begin{quote}
        \mbox{}\\[0.5em]
        $\boxed{\implies}$ Supongamos que existe $v\in \text{Rad}(H)$ distinto del cero, entonces $H(v,w)=v^tHw=0$ para todo $w\in T_pM$, luego $v^tH=0 \text{ si y solo si } v^t = 0$ ya que $H$ es no singular y $v^tH$ es una combinaciﾃｳn lineal las filas de $H$, contradicciﾃｳn.

        $\boxed{\impliedby}$ $\text{Rad}(H)=\{0\}$. Supongamos que $p$ es degenerado y sea $v\in T_pM\setminus\{0\}$, entonces existe $w\in T_pM$ tal que $v^tHw\neq 0$. Asﾃｭ, $v^tH\neq 0$ y como $v$ era arbitrario se tiene que $v^tH\neq 0$ para todo $v\in T_pM\setminus\{0\}$, luego $H$ es no singular y hemos llegado a una contradicciﾃｳn.
    \end{quote}
\end{remark}
\bigskip

\begin{nota}
    De ahora en adelante nos referiremos a la forma bilineal inducida por $f$ como $\text{Hess}_pf$ y a su matriz con respecto de la base del plano de tangente como $H_pf$ o simplemente $H$ si el contexto lo permite. Ademﾃ｡s nos referiremos al ﾃｭndice y la nulidad de $\text{Hess}_pf$ como \textit{ﾃｭndice y nulidad de $f$ en $p$}.
\end{nota}
\bigskip


Ahora que hemos desarrollado el vocabulario para hablar de los puntos crﾃｭticos de una funciﾃｳn sobre los reales, cerramos esta secciﾃｳn con el enunciado y la demostraciﾃｳn del \textit{Lema de Morse}. 

Este resultado es fundamental para el desarrollo de la teorﾃｭa de Morse. A grandes rasgos, establece que en un entorno de un punto crﾃｭtico no degenerado existe un sistema local de coordenadas respecto al cual la funciﾃｳn $f$ puede expresarse como una forma cuadrﾃ｡tica. Desde un punto de vista geomﾃｩtrico, esto implica que $f$ se asemeja a un paraboloide en dicho entorno. Esto resulta increﾃｭblemente ﾃｺtil y constituye la base de los argumentos topolﾃｳgicos que usaremos a partir de este punto.

\bigskip

Antes de enunciar el Lema de Morse necesitamos probar el siguiente resultado de cﾃ｡lculo diferencial.

\begin{lemma}
    Sean $V$ un entorno convexo del origen en $\mathbb{R}^n$ y $f\in \mathcal{C}^\infty(V)$ con $f(0)=0$. Entonces,
    \begin{equation*}
        f(x_1,\dots,x_n)=\sum_{i=1}^n x_ig_i(x_1,\dots,x_n)
    \end{equation*}
    para unas funciones $g_i\in \mathcal{C}^\infty(V)$ que ademﾃ｡s, cumplen que $g_i(0)=\frac{\partial f}{\partial x_i}(0)$.
\end{lemma}
\begin{proof}
    Se tiene
\begin{align*}
    f(x_i,\dots,x_n)=\int_0^1 \frac{df(tx_1,\dots,tx_n)}{dt}dt = \int_0^1 \big(\sum_{i=1}^n \frac{\partial f}{\partial x_i}&(tx_1,\dots,tx_n)x_i \big)dt =\\
    &=\sum_{i=1}^n x_i \left(\int_0^1 \frac{\partial f}{\partial x_i}(tx_1,\dots,tx_n )dt\right)
\end{align*}

Tomamos $g_i(x_1,\dots,x_n) = \int_0^1 \frac{\partial f}{\partial x_i}(tx_1,\dots,tx_n )dt$ y se cumple que $g_i(0)=\int_0^1 \frac{\partial f}{\partial x_i}(0)dt =\frac{\partial f}{\partial x_i}(0)$.

\end{proof}
\bigskip

Vayamos ahora con la demostraciﾃｳn, para la que nos hemos basado tanto en la de Milnor (\cite[p.~6]{milnor1963}) como en la que se encuentra en \cite[p.~431]{marsdenHoffman1993eca}.

\begin{lemma}[Lema de Morse]
\label{lemamorse}
    Sean $p$ un punto crﾃｭtico no degenerado de una funciﾃｳn suave $f:M\to \mathbb{R}$. Entonces existe un sistema de coordenadas local $(y_1,\dots, y_n)$ en un entorno $\mathcal{U}$ de $p$ tal que $y_i(p)=0, \forall i\in \{1\dots,n\}$ y se verifica
    \begin{equation*}
        f=f(p)-(y_1)^2-\dots-(y_\lambda)^2 + (y_{\lambda +1})^2+\dots + (y_n)^2
    \end{equation*}
    en todo $\mathcal{U}$ con $\lambda$ el ﾃｭndice de $f$ en $p$.
\end{lemma}
\bigskip

\begin{proof}[Demostraciﾃｳn del Lema de Morse]
    %################ PARTE 1, INDICE ######################
    En primer lugar, veamos que de poder expresar $f$ de esa manera en $\mathcal{U}$ entonces $\lambda$ es necesariamente el ﾃｭndice de $f$ en $p$. Sea $(x_1,\dots, x_n)$ un sistema de coordenadas local en $\mathcal{U}$ y supongamos que
    \begin{equation*}
        f=f(p)-(x_i)^2-\dots-(x_\lambda)^2 + (x_{\lambda +1})^2+\dots + (x_n)^2.
    \end{equation*}
    Entonces se tiene que 

    \[
        \frac{\partial^2 f}{\partial x_i \partial x_j}=
        \begin{cases}
            -2, ~i=j\le \lambda\\
            ~~~2, ~i=j\ge \lambda\\
            ~~~0, i\neq j
        \end{cases}
    \]

    Es decir, $H_pf$ es una matriz diagonal con $\lambda$ valores negativos y $n-\lambda$ valores positivos. Luego tiene $\lambda$ autovalores negativos, lo que quiere decir que existe un subespacio de dimensiﾃｳn $\lambda$ de $T_pM$ en el que es definida negativa y otro de dimensiﾃｳn $n-\lambda$ en el que es definida positiva.

    La ﾃｺnica forma en que $\lambda$ pudiera ser distinto que el ﾃｭndice de $f$ en $p$ es si existiera un subespacio de dimensiﾃｳn mayor que el mencionado en el que $H_pf$ fuese definida negativa, lo cual es imposible.\\


    %############ PARTE 2, SISTEMA DE COORDENADAS ##################
    Hacen falta algunos preparativos antes de demostrar que el sistema de coordenadas $(y_1,\dots,y_n)$ existe.
    
    En primer lugar podemos asumir sin pﾃｩrdida de generalidad que, en coordenadas, $p$ es el origen de $\mathbb{R}^n$ componiendo la carta escogida con la traslaciﾃｳn que lleva $p$ al origen, que es un difeomorfismo. Tambiﾃｩn asumimos $f(p)=f(0)=0$, de no ser asﾃｭ podrﾃｭamos tomar $\bar{f}(q)=f(q)-f(p)$ que verifica $\bar{f}(p)=0$ y tiene a $p$ como punto crﾃｭtico no degenerado y deshacer despuﾃｩs el cambio. 

    Por el lema anterior $f(x_1,\dots,x_n)=\sum_{j=1}^n x_jg_j(x_1,\dots,x_n)$ en un entorno convexo del origen. Ahora, como $0$ es un punto crﾃｭtico de $f$ se tiene
    \begin{equation*}
        \frac{\partial f}{\partial x_j}(0)=0 ~\text{ y }~ g_j(0)=\int_0^1 \frac{\partial f}{\partial x_j}(0)~dt=0.
    \end{equation*}
    Luego podemos aplicar de nuevo el lema a cada una de las $g_j$ obteniendo
    \begin{equation*}
        g_j(x_1,\dots,x_n)=\sum_{i=1}^n x_ih_{ij}(x_1,\dots,x_n).
    \end{equation*}
    Y, por tanto,
    \begin{equation*}
        f(x_1,\dots,x_n)=\sum_{i,j} x_ix_jh_{ij}(x_1,\dots,x_n).
    \end{equation*}
    %aqui iba lo de quotes
    Ademﾃ｡s, la matriz $H(0)=(h_{ij}(0))$ es igual a $\frac{1}{2}H_0f=\frac{1}{2}(\frac{\partial^2f}{\partial x_i\partial x_j}(0))$ y por lo tanto es simﾃｩtrica y no singular. En efecto, sabemos que 
    \begin{equation*}
        h_{ij}=\int_0^1 \frac{\partial g_i}{\partial x_j}(tx_1,\dots,tx_n)dt ~~\text{ y}~~ g_{i}=\int_0^1 \frac{\partial f}{\partial x_i}(tx_1,\dots,tx_n)dt.
    \end{equation*}
    Luego
    \begin{gather*}
        \frac{\partial g_i}{\partial x_j}(x_1,\dots,x_n)= \int_0^1 \frac{\partial}{\partial x_j}\left(\frac{\partial f}{\partial x_i}(tx_1,\dots,tx_n)\right)dt = \int_0^1 \left(\sum_{k=1}^n\frac{\partial^2 f}{\partial x_i \partial x_k}(tx_1,\dots,tx_n) \frac{\partial(tx_k)}{\partial x_j}\right)dt =\\
        = \int_0^1t\frac{\partial^2 f}{\partial x_i \partial x_j}(tx_1,\dots,tx_n)dt.
    \end{gather*}
    Asﾃｭ, 
    \begin{gather*}
        \frac{\partial g_i}{\partial x_j}(0)=\int_0^1 t\frac{\partial^2 f}{\partial x_i \partial x_j}(0)dt=\frac{\partial^2 f}{\partial x_i \partial x_j}(0)\int_0^1tdt= \frac{1}{2}\frac{\partial^2 f}{\partial x_i \partial x_j}(0)\implies\\
        \implies h_{ij}(0)=\frac{1}{2}\frac{\partial^2 f}{\partial x_i \partial x_j}(0)\int_0^1 dt =\frac{1}{2}\frac{\partial^2 f}{\partial x_i \partial x_j}(0)
    \end{gather*}
    %############### INDUCCIﾃ哲 ###########################
    Finalmente, tenemos $f$ expresada similarmente a una forma cuadrﾃ｡tica. Lo que queremos hacer a continuaciﾃｳn es diagonalizarla, para ello procederemos por inducciﾃｳn. 
    
    El caso base de la inducciﾃｳn ($r=1$) ya estﾃ｡ demostrado.

    Supongamos ahora que existe un sistema de coordenadas $(u_i,\dots,u_n)$ en un entorno $\mathcal{U}_1$ de $0$ quizﾃ｡s mﾃ｡s pequeﾃｱo que $\mathcal{U}$ tal que
    \begin{equation}
        f = \pm (u_1)^2 \pm\dots\pm (u_{r-1})^2 + \sum_{i,j\ge r} u_i u_j H_{ij}(u_1,\dots,u_n)
    \label{eq1}
    \end{equation}
    en todo $\mathcal{U}_1$ con $r > 1$ y $H_{ij}=H_{ji}$ para cualesquiera $i$ y $j$.

    Podemos realizar ahora un cambio lineal de coordenadas en $(u_r,\dots, u_n)$ para diagonalizar $\sum_{i,j\ge r} u_i u_j H_{ij}(0)$.
    
    Sabemos que, por ser la matriz $(H_{ij})$ no singular, los tﾃｩrminos de la diagonal son todos no nulos. Luego podemos asumir $H_{rr}\neq 0$. Sea $g(u_1,\dots,u_n)=|H_{rr}(u_1,\dots,u_n)|^{1/2}$, es claro que existe un entorno $\mathcal{U}_2\subset \mathcal{U}_1$ de $0$ en el que es suave y no se anula. Definimos el nuevo sistema de coordenadas
    \begin{equation}
        \begin{cases}
            V_i(u_1,\dots,u_n) = u_i,~~i\neq r\\
            V_r(u_1,\dots,u_n)=g(u_1,\dots,u_n)\left[u_r+\sum_{i>r}u_i\frac{H_{ir}(u_1,\dots,u_n)}{H_{rr}(u_1,\dots,u_n)}\right],
        \end{cases}
    \label{eq2}
    \end{equation}

    veamos que en efecto lo es. Se tiene que la matriz Jacobiana de la transformaciﾃｳn de coordenadas en el $0$ es
    \begin{equation*}
        \frac{\partial (V_1,\dots, V_n)}{\partial (u_1,\dots, u_n)} = 
        \begin{pmatrix}
            1 & 0 & ~ & \dots & ~ & ~ & 0 \\
            0 & ~ & ~ & ~ & ~ & ~ & 0 \\
            \vdots & ~ & 1 & ~ & ~ & ~ & \vdots \\
            \frac{\partial V_r}{\partial u_1} & \dots & ~ & g(0) & ~ & \dots & \frac{\partial V_r}{\partial u_n} \\
            0 & ~ & ~ & ~ & 1 & ~ & 0 \\
            \vdots & ~ & ~ & ~ & ~ & ~ & \vdots \\
            0 & ~ & ~ & \dots & ~ & 0 & 1
        \end{pmatrix},
    \end{equation*}
    y es no singular, basta restar a la $r$-ﾃｩsima fila las demﾃ｡s multiplicadas por el tﾃｩrmino $\frac{\partial V_i}{\partial u_j}$ adecuado para que sea diagonal, y los tﾃｩrminos de la diagonal son todos no nulos.

    Se sigue entonces del \textit{teorema de la funciﾃｳn inversa} que la transformaciﾃｳn es un difeomorfismo sobre un entorno mﾃ｡s pequeﾃｱo $\mathcal{U}_3\subset \mathcal{U}_2$ del $0$ y por tanto $(v_1,\dots, v_n)$ es un sistema de coordenadas local alrededor de $p$.

    Ahora consideramos
    \begin{equation}
        u_ru_rH_{rr}(u_1,\dots, u_n) + 2\sum_{j=r+1}^n u_ju_rH_{jr}(u_1,\dots, u_n).
    \label{eq3}
    \end{equation}

    Usando la simetrﾃｭa de $H_{ij}$ observamos esta expresiﾃｳn no es mﾃ｡s que los tﾃｩrminos de \eqref{eq1} que tienen $j=r$ o $i=r$. Comparando \eqref{eq1} y \eqref{eq2} llegamos a que \eqref{eq3} es
    \begin{equation}
        \pm V_rV_r - \frac{1}{H_{rr}}\left[ \sum_{i>r} u_iH_{ir}(u_1,\dots, u_n) \right]^2
    \label{eq4}
    \end{equation}

    En efecto,
    \begin{align*}
        V_rV_r =& ~ g^2 \left[u_r+\sum_{i>r}u_i\frac{H_{ir}}{H_{rr}}\right]^2 =\\
        =& ~ g^2 \left( u_r^2 + \left[ \sum_{i>r}u_i\frac{H_{ir}}{H_{rr}}\right]^2 + 2u_r\sum_{i>r}u_i\frac{H_{ir}}{H_{rr}} \right) = \\
        =& ~ |H_{rr}| u_r^2 + \frac{|H_{rr}|}{H_{rr}^2} \left[\sum_{i>r}u_iH_{ir}\right]^2 + 2u_r\frac{|H_{rr}|}{H_{rr}}\sum_{i>r}u_iH_{ir} = \\
        =& ~ \pm H_{rr}u_r^2 \pm \frac{1}{H_{rr}}\left[\sum_{i>r}u_iH_{ir}\right]^2 \pm 2u_r\sum_{i>r}u_iH_{ir},
    \end{align*}

    donde el $\pm$ serﾃ｡ $+$ si $H_{rr}$ es positivo y $-$ si es negativo y de la misma forma en \eqref{eq4}. 
    
    Asﾃｭ, es claro que \eqref{eq3} y \eqref{eq4} son equivalentes. Y desde aquﾃｭ es fﾃ｡cil ver como \eqref{eq1} se convierte en 
    \begin{equation*}
        f = \pm (V_1)^2 \pm\dots\pm (V_{r-1})^2 + (V_r)^2  \pm \sum_{i,j > r} V_iV_j\tilde{H}_{ij}(V_1,\dots,V_n)
    \end{equation*}
    para unos nuevos tﾃｩrminos $\tilde{H}_{ij}$ simﾃｩtricos en funciﾃｳn del nuevo sistema de coordenadas.

    Esto concluye la inducciﾃｳn ya que tenemos un sumando cuadrﾃ｡tico mﾃ｡s y queda probada la existencia del sistema de coordenadas local buscado.
    
\end{proof}

\begin{remark}
    Durante la demostraciﾃｳn queda implﾃｭcita una parte del proceso. Al definir el nuevo sistema de coordenadas en el paso de inducciﾃｳn lo que estamos haciendo es definir una funciﾃｳn que podemos llamar $l:\mathbb{R}^n \to \mathbb{R}^n$ que, dadas las coordenadas $(u_1(p),\dots, u_n(p))$ de un punto $p\in M$, las lleva a unas nuevas $(v_1(p),\dots, v_n(p))$ y es un difeomorfismo entre abiertos de $\mathbb{R}^n$. Asﾃｭ, si la carta que inducﾃｭa el sistema de coordenadas $(u_1,\dots, u_n)$ era $\varphi : M \to \mathbb{R}^n$, la que induce $(v_1,\dots, v_n)$ serﾃ｡ $l\circ \varphi$.
\end{remark}
\bigskip

El siguiente corolario serﾃ｡ clave a la hora de demostrar uno de los teoremas principales de la teorﾃｭa de Morse (ver \ref{cwvariedad}) y muestra una de las ventajas fundamentales de trabajar con puntos crﾃｭticos no degenerados.

\begin{corollary}
        Sea $M$ una variedad diferenciable. Todo punto crﾃｭtico no degenerado de una funciﾃｳn suave de $M$ en $\mathbb{R}$ estﾃ｡ aislado respecto de los demﾃ｡s puntos crﾃｭticos.
\end{corollary}
\begin{proof}
    Sea $p\in M$ un punto crﾃｭtico no degenerado de una funciﾃｳn $f$. Existe un sistema de coordenadas local $(y_1,\dots, y_n)$ en un entorno $\mathcal{U}$ de $p$ tal que $y_i(p)=0,~~\forall i$ y 
    \begin{equation*}
        f=f(p)-(y_i)^2-\dots-(y_\lambda)^2 + (y_{\lambda +1})^2+\dots + (y_n)^2
    \end{equation*}
    Por el Lema de Morse (\ref{lemamorse}).\\

    Supongamos que existe otro punto crﾃｭtico $q\in\mathcal{U}$, entonces
    \begin{equation*}
        \frac{\partial f}{\partial x_i}(q)=\pm 2q_i= 0\iff q_i= y_i(q)=0\iff q_i=p_i
    \end{equation*}
    Luego $p$ es el ﾃｺnico punto crﾃｭtico de f en $\mathcal{U}$.
    
\end{proof}

A continuaciﾃｳn presentamos algunos puntos crﾃｭticos degenerados y no degenerados junto con las grﾃ｡ficas de sus funciones correspondientes. Obsﾃｩrvese que mientras los puntos no degenerados han de ser aislados, una funciﾃｳn puede tener puntos crﾃｭticos degenerados tanto aislados como no aislados.

\begin{example}
    \hfil
    \begin{enumerate}
        \item La funciﾃｳn $f(x,y)=(x^2-1)^2 + (y^2-1)^2$ tiene nueve puntos crﾃｭticos no degenerados. Es fﾃ｡cil comprobar que $\text{Hess}_pf$ es una matriz diagonal cuyos elementos no se anulan en ninguno de los puntos crﾃｭticos. 
        \item La funciﾃｳn $g(x,y)=x^3-3xy^2$ cuya grﾃ｡fica se corresponde con la ``silla de mono'' posee un ﾃｺnico punto crﾃｭtico degenerado en el punto $(0,0)$. Basta ver que $\text{Hess}_{(0,0)}f$ tiene ceros en las cuatro entradas.
        \item Observamos que $h(x,y)= x^2y^2$ tiene infinitos puntos crﾃｭticos degenerados, los de la forma $(0,y)$ y $(x,0)$.
    \end{enumerate}
\end{example}
