
Cuando Morse desarrolló su teoría, a lo largo de los años veinte \cite{BottMarstonMorse}, la escena matemática era muy distinta; no fue hasta $1949$ que Whitehead formalizó siquiera el concepto de CW-complejo tal y como lo conocemos \cite{KletteCellComplexesHistory}. En particular, el teorema \ref{cwvariedad} aún no había sido demostrado, por lo que la relación entre la topología de una variedad diferenciable y los puntos críticos de una función de Morse fue descrita mediante una colección de desigualdades. 

En este capítulo presentamos esta formulación original siguiendo de nuevo el desarrollo encontrado en \cite{milnor1963}. 

\section{Subaditividad}

Empezamos introduciendo algunos resultados de funciones subaditivas.

A lo largo de este capítulo nos referiremos a pares de espacios $(X,Y)$. Conviene aclarar que no se trata de pares de espacios arbitrarios, sino de pares tales que $Y$ es un subespacio de $X$ para el que tiene sentido construir los grupos de homología relativa (ver sección \ref{sec:hom_relat}). Recordemos también que en el caso en el que $Y$ es vacío o contractible recuperamos la homología de $X$.

\begin{definition}
    Sea $S$ una función que toma pares $(X,Y)$ de espacios topológicos y les asigna un entero $S(X,Y)\in \mathbb{Z}$. Decimos que $S$ es \textit{subaditiva} si siempre que $X\supset Y\supset Z$ se tiene que
    \begin{equation*}
        S(X,Z) \le S(X,Y) + S(Y,Z).
    \end{equation*}
    Si en las misma hipótesis se verifica siempre la igualdad diremos que $S$ es \textit{aditiva}.
\end{definition}

\begin{example}
\label{betti_subad}
    \hfill
    \begin{enumerate}[label=(\roman*)]
        \item La función $b_k(X,Y)$ que asigna a cada par $(X,Y)$ su $k$-ésimo número de Betti es subaditiva (siempre que los números de Betti involucrados sean finitos). En efecto, sean $X$, $Y$ y $Z$ tales que $X\supset Y\supset Z$, se tiene la siguiente secuencia exacta:
        \begin{align*}
            \cdots \longrightarrow H_k(Y,Z) \overset{f}{\longrightarrow} H_k(X,Z) \overset{g}{\longrightarrow} H_k(X,Y) \overset{h}{\longrightarrow} \cdots
        \end{align*}
        Esta surge de generalizar la secuencia mencionada en \ref{exact_sec_hom_rel} a una terna \cite[p.~118]{hatcher2002}. De ella se deduce la subaditividad, veamos cómo. Dado un homomorfismo entre grupos abelianos de rango finito $\varphi: G\to H$, por el primer teorema de isomorfía, $G/\text{ker}\varphi$ es isomorfo a $\text{im} \varphi$, luego se tiene
        \begin{equation*}
            \text{rank } G = \text{rank}(\text{ker } \varphi) + \text{rank}(\text{im } \varphi).
        \end{equation*}
        En nuestro caso, tenemos las siguientes igualdades
        \begin{equation*}
            \text{im } f = \text{ker } g ~~\text{ y }~~ \text{im } g = \text{ker } h.
        \end{equation*}
        Además,
        \begin{align*}
            &b_k(Y,Z) = \text{rank}(\text{ker } f) + \text{rank}(\text{im } f) = \text{rank}(\text{ker } f) + \text{rank}(\text{ker } g),\\
            &b_k(X,Z) = \text{rank}(\text{ker } g) + \text{rank}(\text{im } g),\\
            &b_k(X,Y) = \text{rank}(\text{ker } h) + \text{rank}(\text{im } h) = \text{rank}(\text{im } g) + \text{rank}(\text{im } h). 
        \end{align*}
        Por lo que
        \begin{equation*}
            b_k(X,Z) - b_k(Y,Z) = \text{rank}(\text{im } g) - \text{rank}(\text{ker } f) \le b_k(X,Y).
        \end{equation*}
        De donde se deduce la subaditividad.
        \item Aunque la demostración es más elaborada y no la añadiremos, se tiene que la función que asigna a cada par su característica de Euler-Poincaré $\chi (X,Y)$ es aditiva.
    \end{enumerate}
\end{example}

\begin{lemma}
\label{lema__weak_desig_morse}
    Sea $S$ una función subaditiva y sean $X_0\subset \dots \subset X_n$. Entonces,
    \begin{equation*}
        S(X_n, X_0) \le \sum^n_{i=1} S(X_i,X_{i-1}).
    \end{equation*}
    Si $S$ es aditiva se verifica la igualdad.
\end{lemma}
\begin{proof}
    Procedemos por inducción sobre $n$. Para $n=1$ se verifica la igualdad trivialmente y para $n=2$ se tiene la definición de subaditividad. Supongamos que se verifica para $n-1$, entonces
    \begin{equation*}
        S(X_{n-1}, X_0) \le \sum^{n-1}_{i=1} S(X_i,X_{i-1}).
    \end{equation*}
    Utilizando la subaditividad de $S$ y la hipótesis de inducción se tiene
    \begin{equation*}
        S(X_n, X_0) \le S(X_{n-1}, X_0) + S(X_n, X_{n-1}) \le \sum^n_{i=1} S(X_i,X_{i-1}). 
    \end{equation*}

\end{proof}

\begin{lemma}
\label{lema_desig_morse}
    La función $S_{k}$ es subaditiva, donde 
    \begin{equation*}
        S_{k}(X,Y) = \sum_{i=0}^{k} (-1)^i b_{k-i}(X,Y).
    \end{equation*}
\end{lemma}
\begin{proof}
    Supongamos que tenemos la siguiente secuencia exacta
    \begin{align*}
        \cdots \overset{f_{k+1}}{\longrightarrow} A_k \overset{f_{k}}{\longrightarrow} A_{k-1} \overset{f_{k-1}}{\longrightarrow} A_{k-2} \overset{f_{k-2}}{\longrightarrow} \cdots \overset{f_{1}}{\longrightarrow} A_0 \overset{f_{0}}{\longrightarrow} 0.
    \end{align*}
    Recordamos que se define el rango de cada $f_i$ como el rango de $\text{im } f_i$. Así,
    \begin{align*}
        &\text{rank } f_{k+1} = \text{rank } (\text{im }f_{k+1}),\\
        &\text{rank }A_k = \text{rank } (\text{ker }f_{k}) + \text{rank } (\text{im }f_{k}) = \text{rank } (\text{im }f_{k+1}) + \text{rank } f_{k}. 
    \end{align*}
    De lo que se sigue que
    \begin{equation*}
        \text{rank } f_{k+1}  = \text{rank }A_k - \text{rank } f_{k}.
    \end{equation*}
    Si continuamos desarrollando los rangos de las funciones de la misma forma llegamos a lo siguiente
    \begin{align*}
        \text{rank } f_{k+1}  &= \text{rank }A_k - \text{rank } f_{k}=\\
        &= \text{rank }A_k - \text{rank }A_{k-1} + \text{rank } f_{k-1} =\\
        &= \text{rank }A_k - \text{rank }A_{k-1} + \text{rank }A_{k-2} - \text{rank } f_{k-2} =\\
        &~\vdots\\
        &= \text{rank }A_k - \text{rank }A_{k-1} + \text{rank }A_{k-2} - \dots \pm \text{rank }A_0.
    \end{align*}
    Es claro que la última expresión es mayor que cero. Consideramos ahora la secuencia exacta de la terna $X\supset Y\supset Z$ (ver \ref{betti_subad}(i)),
    \begin{align*}
        \cdots \longrightarrow H_{k+1}(Y,Z) \longrightarrow H_{k+1}(X,Z) \longrightarrow H_{k+1}(X,Y) \overset{f}{\longrightarrow} H_{k}(Y,Z) \longrightarrow \cdots
    \end{align*}
    y aplicamos a $f$ lo que acabamos de probar. Observamos que
    \begin{equation*}
        \text{rank} f = b_k(Y,Z) - b_k(X,Z) + b_k(X,Y) - b_{k-1}(Y,Z) + \dots \ge 0.
    \end{equation*}
    Agrupando los términos adecuadamente llegamos a
    \begin{equation*}
        S_{k}(Y,Z) - S_{k}(X,Z) + S_{k}(X,Y) \ge 0,
    \end{equation*}
    lo que concluye la demostración.

\end{proof}

\section{Las desigualdades}
\label{sec:desigmorse}
\begin{theorem}[Desigualdades de Morse, versión débil]
\label{weak_morse_desig}
    Sean $f$ una función de Morse sobre una variedad compacta $n$-dimensional $M$  y $C_{\lambda}$ el número de puntos críticos de índice $\lambda$ de $f$. Entonces,
    \begin{equation}
    \label{weak_morse_desig1}
    \tag{$1_{\lambda}$}
        b_{\lambda}(M) \le C_{\lambda},~\text{ para todo } \lambda\in\Lambda, ~~\text{ y }~~ 
    \end{equation}
    \begin{equation}
    \label{weak_morse_desig2}
    \tag{2}
        \chi(M) = \sum_{\lambda=0}^{n} (-1)^{\lambda} C_{\lambda}.
    \end{equation}
\end{theorem}
\begin{proof}
    Por ser $M$ compacta $f$ tiene una cantidad finita de puntos críticos. Sean $a_1<\dots<a_k$ tales que para todo $i\in \{1,\dots,k\}$ se tiene que $M^{a_i}$ contiene exactamente $i$ puntos críticos, y $M^{a_k}=M$. Entonces,
    \begin{align*}
        H_j(M^{a_i},M^{a_i-1}) = H_j(M^{a_i-1}\cup e^{\lambda_i},M^{a_i-1}) = H_j(e^{\lambda_i},\partial e^{\lambda_i}) = 
        \begin{cases}
            \mathbb{Z} &~~\text{ si }~~ j=\lambda_i,\\
            0 &~~\text{ en cualquier otro caso}.
        \end{cases}
    \end{align*}
    La primera igualdad se sigue del teorema \ref{pegadocelula}, donde $\lambda_i$ es el índice del punto crítico. La segunda se obtiene al aplicar el teorema de escisión (\ref{escision}) al espacio $M^{a_i-1}\cup e^{\lambda_i}$ y los subespacios $M^{a_i-1}$ y $e^{\lambda_i}$. La tercera igualdad quedó demostrada en el ejemplo \ref{hom_relat_disco}.

    Ahora, aplicando el lema \ref{lema__weak_desig_morse} con $M^{a_0}=\emptyset \subset \dots \subset M^{a_k}=M$ y $S=b_{\lambda}$ se tiene 
    \begin{equation*}
        b_{\lambda}(M) \le \sum_{i=1}^k b_{\lambda}(M^{a_i},M^{a_i-1}) = C_{\lambda}.
    \end{equation*}
    Si por el contrario, aplicamos el lema con $S=\chi$ se llega a 
    \begin{gather*}
        \chi(M) = \sum_{i=1}^k \chi(M^{a_i},M^{a_i-1}) = \sum_{i=1}^k \Big(\sum_{j=0}^n (-1)^j b_j(M^{a_i},M^{a_i-1})\Big) = \sum_{j=0}^n (-1)^j \sum_{i=1}^k b_j(M^{a_i},M^{a_i-1})=\\
        = \sum_{j=0}^n (-1)^{j} C_{j}
    \end{gather*}

\end{proof}

Hemos demostrado una primera relación, el rango del $\lambda$-ésimo grupo de homología de $M$ ha de ser menor o igual que el número de puntos críticos de índice $\lambda$ de cualquier función de Morse definida sobre $M$. Además, dada una función de Morse, basta hallar el número de puntos críticos con cada índice posible para conocer su característica de Euler-Poincaré.

Ahora, se pueden obtener desigualdades ligeramente más fuertes que estas:

\begin{theorem}[Desigualdades de Morse]
    Sea $f$ una función de Morse sobre una variedad compacta $n$-dimensional $M$ y $S_{\lambda}$ la función definida en el lema \ref{lema_desig_morse}, para todo $\lambda\in \mathbb{Z}^+$ se verifican las siguientes desigualdades
    \begin{equation*}
        S_{\lambda}(M)\le \sum_{i=1}^k S_{\lambda}(M^{a_i},M^{a_i-1}) = C_{\lambda}-C_{\lambda-1}+\dots\pm C_0,
    \end{equation*}
    donde $(M^{a_i},M^{a_i-1})$ son como en la demostración de \ref{weak_morse_desig}. Equivalentemente,
    \begin{equation}
    \label{desig_morse_fuerte}
    \tag{$3_{\lambda}$}
        b_{\lambda}(M)-b_{\lambda-1}+\dots\pm b_0(M) \le C_{\lambda}-C_{\lambda-1}+\dots\pm C_0.
    \end{equation}
\end{theorem}
\begin{proof}
    Basta aplicar la subaditividad de $S_{\lambda}$ y el lema \ref{lema__weak_desig_morse} con los espacios $M^{a_=0}=\emptyset \subset \dots \subset M^{a_k}=M$.

\end{proof}

Es claro que está colección de desigualdades es más fuerte que la anterior. En efecto, basta observar que sumando (\ref{desig_morse_fuerte}) y ($3_{\lambda-1}$) se obtiene la desigualdad (\ref{weak_morse_desig1}). Por otro lado, si $\lambda$ es mayor que $n$ se tiene que (\ref{desig_morse_fuerte}) es la igualdad (\ref{weak_morse_desig2}).

\bigskip
Veamos una forma de utilizar estas desigualdades. Supongamos que $C_{\lambda+1}=0$, entonces, necesariamente $b_{\lambda+1}=0$. Así, (\ref{desig_morse_fuerte}) y ($3_{\lambda+1}$) quedan
\begin{align*}
    &b_{\lambda}-b_{\lambda-1}+\dots\pm b_0 \le C_{\lambda}-C_{\lambda-1}+\dots\pm C_0,\\
    &-b_{\lambda}+b_{\lambda-1}-\dots\pm b_0(M) \le -C_{\lambda}+C_{\lambda-1}-\dots\pm C_0.
\end{align*}
De lo que se deduce
\begin{equation*}
    b_{\lambda}(M)-b_{\lambda-1}+\dots\pm b_0(M) = C_{\lambda}-C_{\lambda-1}+\dots\pm C_0.
\end{equation*}
Supongamos ahora que $C_{\lambda-1}=0$, se tiene entonces que $b_{\lambda-1}=0$, y de la misma forma se llega a la igualdad
\begin{equation*}
    b_{\lambda-2}(M)-b_{\lambda-3}+\dots\pm b_0(M) = C_{\lambda-2}-C_{\lambda-3}+\dots\pm C_0.
\end{equation*}
Restando ahora esta a la primera igualdad se obtiene que $b_{\lambda} = C_{\lambda}$. Acabamos de demostrar el siguiente corolario.

\begin{corollary}
    Si $C_{\lambda+1} = C_{\lambda-1} = 0$ entonces $b_{\lambda} = C_{\lambda}$ y $b_{\lambda+1} = b_{\lambda-1} = 0$.
\end{corollary}
